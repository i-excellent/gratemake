SELECT  `p`.`h1_ru`, 
        `p`.`h1_de`, 
		  `p`.`sid`, 
		  `i`.`img_de`, 
		  `b`.`brand`, 
		  `a`.`art` , 
 	 MIN(`s`.`price`),
		  `s`.`oldprice`
FROM `product_sid` `p` 
LEFT JOIN `sys_image`    `i` ON `i`.`sid`            = `p`.`sid` 
LEFT JOIN `sys_brand`    `b` ON `b`.`id`             = `p`.`sys_brand_id`
LEFT JOIN `product_art`  `a` ON `a`.`product_sid_id` = `p`.`sid`
LEFT JOIN `product_size` `s` ON `s`.`product_art_id` = `a`.`art`
GROUP BY `p`.`sid`
LIMIT 24
